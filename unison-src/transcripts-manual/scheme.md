print txt =
  match putBytes.impl stdOut (toUtf8 txt) with

prints n f = print (join " " f n ++ "\n")

join sep f =
  use Text ++
  loop acc = cases
    0 -> acc
    n -> loop (!f ++ (sep ++ acc)) (drop n 1)

  cases
    0 -> ""
