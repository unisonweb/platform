  two : Nat -> Nat ->{A} Nat

ha : Request {A} r -> r
ha = cases
  { x } -> x
  { one i -> c } -> handle c (i+1) with ha
  { two i j -> c } -> handle c (i+j) with ha
  { three i j k -> c } -> handle c (i+j+k) with ha
  { four i -> c } -> handle c (j k l -> i+j+k+l) with ha
